### State Management

Using Zustand for lightweight, performant state:

```tsx
import { create } from 'zustand';

interface WindowState {
  windows: Window[];
  activeWindow: string | null;
  openWindow: (app: string) => void;
  closeWindow: (id: string) => void;
  focusWindow: (id: string) => void;
}

const useWindowStore = create<WindowState>((set) => ({
  windows: [],
  activeWindow: null,
  openWindow: (app) => set((state) => ({
    windows: [...state.windows, createWindow(app)],
  })),
  closeWindow: (id) => set((state) => ({
    windows: state.windows.filter((w) => w.id !== id),
  })),
  focusWindow: (id) => set({ activeWindow: id }),
}));
```

## AI Integration

### Google Gemini Setup

<Steps>
  <Step title="Get API Key">
    Sign up for Google AI Studio and get your Gemini API key:
    [https://ai.google.dev](https://ai.google.dev)
  </Step>
  <Step title="Set Environment Variable">
    Create `.env.local`:
    ```bash
    GOOGLE_GEMINI_API_KEY=your_api_key_here
    ```

  </Step>
  <Step title="Initialize Client">
    ```ts
    import { GoogleGenerativeAI } from '@google/generative-ai';
    
    const genAI = new GoogleGenerativeAI(
      process.env.GOOGLE_GEMINI_API_KEY!
    );
    const model = genAI.getGenerativeModel({ 
      model: 'gemini-pro' 
    });
    ```
  </Step>
</Steps>

## Animation Patterns

### Window Animations

```tsx
import { motion } from 'framer-motion';

const Window = ({ children, isActive }) => (
  <motion.div
    initial={{ opacity: 0, scale: 0.9, y: 50 }}
    animate={{ 
      opacity: 1, 
      scale: 1, 
      y: 0,
      zIndex: isActive ? 100 : 1
    }}
    exit={{ 
      opacity: 0, 
      scale: 0.8,
      transition: { duration: 0.2 }
    }}
    drag
    dragConstraints={{ top: 0, left: 0, right: 1000, bottom: 600 }}
    dragElastic={0.1}
    className="window"
  >
    {children}
  </motion.div>
);
```

## Customization

### Adding New Apps

Create a new app in the `app/apps/` directory:

```tsx
// app/apps/calculator/page.tsx
export default function Calculator() {
  return (
    <div className="p-4">
      <h1>Calculator</h1>
      {/* Your app UI */}
    </div>
  );
}
```

Register it in the app registry:

```ts
// lib/apps.ts
export const apps = [
  {
    id: 'calculator',
    name: 'Calculator',
    icon: 'calculator',
    path: '/apps/calculator'
  },
  // ... other apps
];
```
